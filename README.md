Simple Emitter
==============

Just a simple event emitter

## Install

    npm install simple-emitter


## Usage

    var AudioControls = require('audio-controls');
    var audioControls = new AudioControls(
        {
            'playQueue': playQueue,
            'playPause': '#control-play-pause',
            'previous': '#control-previous',
            'next': '#control-next'
        }
    );
